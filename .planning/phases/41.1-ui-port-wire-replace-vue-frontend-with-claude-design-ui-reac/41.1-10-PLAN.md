<objective>
Two compound fixes:

**NR-7 (Advisory dies on system sleep)** — User reported: started advisory synthesis (~5 min job), put M1 to sleep, woke. Advisory stuck forever on "SYNTHESIZING ADVISORY…" because Ollama unloaded the 35B model on sleep (RAM 48GB → 29.77GB). Fix has two halves:
  a. Prevent the sleep: backend wraps the synthesis block in `caffeinate -i` (idle-sleep inhibitor). When request finishes (success OR exception), the caffeinate subprocess is terminated in the finally block.
  b. Detect when prevention fails (e.g., user lid-close, manual sleep): GET /api/health/ollama exposes connection state; frontend polls during active tasks; TaskBanner surfaces "OLLAMA DISCONNECTED" with a Retry button.

**NR-8 (No long-task UX affordance)** — User had no way to distinguish "running" from "dead/stuck" during 54-min sim or 5+ min advisory. Fix: a TaskBanner component (mounted in app_v2.jsx) shows a sticky strip below the topbar whenever a long-running task is active. Surfaces: task name, elapsed time (mm:ss), backend health (when relevant), and a Cancel button (simulation only — advisory has no abort endpoint, status-only).

Output: User can sleep their M1 mid-advisory and the synthesis survives (caffeinate); if it doesn't survive (edge case), the UI surfaces the disconnection within 5 seconds with a clear retry path; user has continuous task-state visibility during long jobs.
</objective>

    ```python
    async def _run_advisory_synthesis(
        app_state: AppState,
        cycle_id: str,
        portfolio: PortfolioSnapshot,
    ) -> None:
        """Background task: caffeinate-guarded orchestrator synthesis (NR-7).

        Wraps the long-running synthesize() call with a macOS `caffeinate -i`
        subprocess to prevent idle-sleep from unloading the 35B orchestrator
        model mid-job. The caffeinate handle is terminated in the finally
        block — no orphan subprocesses on success OR exception.
        """
        orchestrator = app_state.settings.ollama.orchestrator_model_alias
        model_manager = app_state.model_manager
        gm = app_state.graph_manager
        ollama_client = app_state.ollama_client
        assert model_manager is not None
        assert gm is not None
        assert ollama_client is not None

        # NR-7: idle-sleep inhibitor (macOS). On Linux/CI, caffeinate is absent;
        # we log + continue (the test suite + Linux dev paths must not break).
        caffeinate_proc: subprocess.Popen | None = None
        if shutil.which("caffeinate") is not None:
            try:
                caffeinate_proc = subprocess.Popen(
                    ["caffeinate", "-i"],
                    stdout=subprocess.DEVNULL,
                    stderr=subprocess.DEVNULL,
                )
                log.info("advisory_caffeinate_started", cycle_id=cycle_id, pid=caffeinate_proc.pid)
            except (FileNotFoundError, OSError) as exc:
                log.warning(
                    "advisory_caffeinate_unavailable",
                    cycle_id=cycle_id,
                    error=str(exc),
                )
                caffeinate_proc = None
        else:
            log.warning("advisory_caffeinate_skipped", cycle_id=cycle_id, reason="binary_missing")

        try:
            await model_manager.load_model(orchestrator)

            report_obj = await synthesize(
                cycle_id=cycle_id,
                portfolio=portfolio,
                graph_manager=gm,
                ollama_client=ollama_client,
                orchestrator_model=orchestrator,
            )

            await aiofiles.os.makedirs(ADVISORY_DIR, exist_ok=True)
            output_path = ADVISORY_DIR / f"{cycle_id}_advisory.json"
            async with aiofiles.open(output_path, "w", encoding="utf-8") as f:
                await f.write(report_obj.model_dump_json(indent=2))

            log.info(
                "advisory_generation_complete",
                cycle_id=cycle_id,
                output=str(output_path),
                affected=report_obj.affected_holdings,
            )
        except Exception as exc:
            log.error("advisory_generation_failed", cycle_id=cycle_id, error=str(exc))
            raise
        finally:
            try:
                await model_manager.unload_model(orchestrator)
            except Exception:
                log.warning("orchestrator_unload_failed_after_advisory", cycle_id=cycle_id)
            # NR-7: always terminate caffeinate (even on raise).
            if caffeinate_proc is not None:
                try:
                    caffeinate_proc.terminate()
                    caffeinate_proc.wait(timeout=2.0)
                    log.info("advisory_caffeinate_terminated", cycle_id=cycle_id)
                except Exception as exc:
                    log.warning(
                        "advisory_caffeinate_terminate_failed",
                        cycle_id=cycle_id,
                        error=str(exc),
                    )
    ```

    ```python
    """Health check endpoints (NR-7 detection layer).

    Exposes simple readiness probes the frontend can poll without subscribing
    to /ws/state. Used by useOllamaHealth to detect orchestrator-model
    disconnections after macOS sleep.
    """
    from __future__ import annotations

    from typing import TYPE_CHECKING

    import structlog
    from fastapi import APIRouter, Request
    from pydantic import BaseModel

    if TYPE_CHECKING:
        from alphaswarm.app import AppState

    log = structlog.get_logger(component="web.health")

    router = APIRouter()


    class OllamaHealthResponse(BaseModel):
        connected: bool
        models_loaded: list[str]


    @router.get("/health/ollama", response_model=OllamaHealthResponse)
    async def health_ollama(request: Request) -> OllamaHealthResponse:
        """Return Ollama connection status + currently-loaded models.

        Used by the frontend TaskBanner (NR-7) to detect when the orchestrator
        model unloaded after macOS sleep, so the UI can surface a retry CTA.

        NEVER raises — always returns 200 with disconnected=true on failure.
        """
        app_state: AppState = request.app.state.app_state
        ollama_client = app_state.ollama_client
        if ollama_client is None:
            return OllamaHealthResponse(connected=False, models_loaded=[])

        try:
            # Ollama python client exposes list() returning {'models': [{'name': ..., ...}]}
            # Confirm exact method via Read on src/alphaswarm/ollama_client.py before writing.
            models_response = await ollama_client.list()
            models_field = getattr(models_response, "models", None) or models_response.get("models", [])
            names = [getattr(m, "name", None) or m.get("name", "") for m in models_field]
            return OllamaHealthResponse(connected=True, models_loaded=[n for n in names if n])
        except Exception as exc:
            log.warning("health_ollama_query_failed", error=str(exc))
            return OllamaHealthResponse(connected=False, models_loaded=[])
    ```

    ```ts
    import { useEffect, useState } from 'react';
    import { getOllamaHealth, type OllamaHealth } from '../api/health';

    export interface UseOllamaHealthResult {
      health: OllamaHealth | null;
      error: Error | null;
    }

    /**
     * NR-7: poll /api/health/ollama every 5s while `active` is true.
     * Returns null health when inactive; never polls when inactive.
     */
    export function useOllamaHealth(active: boolean): UseOllamaHealthResult {
      const [health, setHealth] = useState<OllamaHealth | null>(null);
      const [error, setError] = useState<Error | null>(null);

      useEffect(() => {
        if (!active) {
          setHealth(null);
          setError(null);
          return;
        }
        let cancelled = false;
        const tick = () => {
          getOllamaHealth()
            .then((h) => { if (!cancelled) { setHealth(h); setError(null); } })
            .catch((e: unknown) => {
              if (cancelled) return;
              setHealth({ connected: false, models_loaded: [] });
              setError(e instanceof Error ? e : new Error(String(e)));
            });
        };
        tick();
        const id = window.setInterval(tick, 5000);
        return () => {
          cancelled = true;
          window.clearInterval(id);
        };
      }, [active]);

      return { health, error };
    }
    ```

    ```tsx
      const mm = Math.floor(s / 60).toString().padStart(2, '0');
      const ss = (s % 60).toString().padStart(2, '0');
      return `${mm}:${ss}`;

    export function TaskBanner({
    ```
