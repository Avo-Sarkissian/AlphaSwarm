    ```ts
    import { useTelemetry }  from '../context/TelemetryContext';
    import { useConnection } from '../context/ConnectionContext';

    export type OverlayKind = 'idle' | 'seeding' | 'mempause' | 'error' | null;

    export interface OverlayState {
      kind: OverlayKind;
      memPct?: number;
    }

    /** Decision table matches RESEARCH §states.jsx wiring. null = live graph shows. */
    export function useLifecycleOverlay(memThresholdPct: number = 90): OverlayState {
      const tel  = useTelemetry();
      const conn = useConnection();

      if (conn.reconnectFailed) return { kind: 'error' };

      const phase = tel.phase;
      if (phase === 'idle')    return { kind: 'idle' };
      if (phase === 'seeding') return { kind: 'seeding' };

      const mem = tel.telemetry?.memMb ?? 0;
      if (mem >= memThresholdPct) return { kind: 'mempause', memPct: mem };

      return { kind: null };
    }
    ```
