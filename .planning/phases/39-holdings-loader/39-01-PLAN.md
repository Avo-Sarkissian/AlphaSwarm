```python
from __future__ import annotations

import csv

from decimal import Decimal, InvalidOperation

import structlog

from alphaswarm.holdings.types import Holding, PortfolioSnapshot
from alphaswarm.security.hashing import sha256_first8

class HoldingsLoadError(Exception):
    """Raised when the CSV cannot be parsed into a PortfolioSnapshot.

    Caught by the FastAPI lifespan in Plan 02 so missing or malformed holdings
    data surfaces as a structured 503 instead of crashing the WebSocket loop
    (D-08, Phase 39 success criterion 4).

def _as_of_from_path(path: Path) -> datetime:
    """Return tz-aware UTC datetime for `as_of`, with a race-safe fallback.

    Research Pattern 5: Path.stat().st_mtime converted to UTC reflects when
    the portfolio was last exported. If stat() raises after the file was
    successfully read (NFS race, permissions change), fall back to the current
    UTC time rather than letting the error propagate.
    """
    try:
        mtime = path.stat().st_mtime
        return datetime.fromtimestamp(mtime, tz=UTC)
    except OSError:
        log.warning("holdings_stat_fallback", path=str(path))
        return datetime.now(UTC)


class HoldingsLoader:
    """Loads a Schwab CSV export into a PortfolioSnapshot (HOLD-01, HOLD-02).

    Stateless — the single public method is a classmethod. The loader is pure
    Python stdlib (csv, decimal, pathlib, datetime) plus the shared
    sha256_first8 helper. It performs synchronous file I/O; the caller
    (FastAPI lifespan) invokes it at startup so the event loop is never
    blocked at request time.
    """

    @classmethod
    def load(cls, path: Path) -> PortfolioSnapshot:
        """Read the CSV at `path` and return a merged PortfolioSnapshot.

        Raises:
            HoldingsLoadError: on any failure (missing file, missing column,
                invalid numeric, empty data body, stat failure, etc.).
        """
        try:
            with path.open(newline="", encoding="utf-8") as f:
                reader = csv.DictReader(f)
                if reader.fieldnames is None:
                    raise HoldingsLoadError("CSV is empty — no header row")
                missing = REQUIRED_COLUMNS - set(reader.fieldnames)
                if missing:
                    raise HoldingsLoadError(
                        f"CSV missing columns: {sorted(missing)}"
                    )
                rows = list(reader)
        except FileNotFoundError as exc:
            raise HoldingsLoadError(f"CSV not found: {path}") from exc
        except OSError as exc:
            raise HoldingsLoadError(f"Cannot read CSV: {exc}") from exc

        if not rows:
            # Pitfall 2: sha256_first8("") raises TypeError. Guard before hashing.
            raise HoldingsLoadError("CSV has no data rows")

        holdings_list: list[Holding] = []
        account_labels: set[str] = set()
        for row in rows:
            try:
                qty = Decimal(row["shares"])
                cost_per_share = Decimal(row["cost_basis_per_share"])
            except InvalidOperation as exc:
                raise HoldingsLoadError(
                    f"Invalid numeric value in row {row!r}: {exc}"
                ) from exc
            account_labels.add(row["account"])
            holdings_list.append(
                Holding(
                    ticker=row["symbol"],
                    qty=qty,
                    cost_basis=qty * cost_per_share,  # D-02 — total position cost
                )
            )

        # HOLD-02: account_number_hash is the sha256_first8 digest of the
        # sorted unique account labels joined with '|'. With the real CSV this
        # is sha256_first8("individual|roth_ira"). Guarded above: rows is
        # non-empty, so account_labels is non-empty, so the join is non-empty.
        account_hash = sha256_first8("|".join(sorted(account_labels)))

        as_of = _as_of_from_path(path)

        # Pitfall 7: NEVER log raw Holding fields (ticker/qty/cost_basis).
        # Count + sorted account labels only — the labels (individual/roth_ira)
        # are not raw account numbers, so logging them is safe.
        log.info(
            "holdings_loaded",
            path=str(path),
            count=len(holdings_list),
            accounts=sorted(account_labels),
        )

        return PortfolioSnapshot(
            holdings=tuple(holdings_list),  # PortfolioSnapshot requires tuple
            as_of=as_of,
            account_number_hash=account_hash,
        )
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Filesystem → HoldingsLoader | `settings.holdings_csv_path` (env-controlled) crosses from untrusted config into the loader. A malformed CSV cell, a truncated file, or a permissions race must convert to a typed domain exception; nothing uncontrolled (InvalidOperation, OSError, TypeError) may escape. |

| HoldingsLoader → structlog | Log emissions are a sink that could leak Holding fields into disk/centralized logs if a developer adds `log.info("x", holdings=snap.holdings)`. The ISOL-07 canary enforces this at the simulation boundary; this plan adds a grep-invariant at the loader boundary. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-39-01 | Tampering | CSV cells parsed by `Decimal(row["shares"])` | mitigate | `InvalidOperation` is caught and re-raised as `HoldingsLoadError` with a structured message. Non-numeric or overflow values cannot crash the loader; Task 3 `test_load_malformed_csv_invalid_numeric` verifies the chain (`__cause__ is not None`). Row context is included only in the exception message, NOT in log output, so a malformed row cannot leak into persistent logs. |
| T-39-02 | Information Disclosure | Holding field values in log output | mitigate | `log.info("holdings_loaded", ...)` emits ONLY `path`, `count`, `sorted(account_labels)` — never ticker/qty/cost_basis. Enforced by grep invariant `test_loader_module_does_not_log_holding_field_values` in Task 3 and by the ISOL-07 canary in Phase 37 (which Plan 02 re-runs). Raw account numbers are never in scope because the CSV uses labels (`individual`/`roth_ira`) that are not account identifiers. |
| T-39-03 | Denial of Service | sha256_first8("") raises TypeError on empty label set | mitigate | `if not rows: raise HoldingsLoadError("CSV has no data rows")` runs BEFORE the sha256_first8 call (Pitfall 2). Task 3 `test_load_empty_body_header_only` verifies the typed exception is raised — NOT the underlying TypeError. Without this guard, a zero-row CSV would crash the FastAPI lifespan in Plan 02 with a raw TypeError, violating D-08. |
| T-39-04 | Denial of Service | Race between path.open() success and path.stat() failure | mitigate | `_as_of_from_path` wraps `path.stat()` in a `try/except OSError` and falls back to `datetime.now(UTC)` with a `log.warning("holdings_stat_fallback", path=str(path))`. Task 3 does not explicitly exercise the race (hard to simulate) but the grep acceptance criteria assert the fallback branch exists in source. |
| T-39-05 | Tampering | Path traversal via `ALPHASWARM_HOLDINGS_CSV_PATH` env var | accept (scope deferred to Plan 02) | Plan 02 owns the AppSettings field. | T-39-06 | Information Disclosure | Exception messages leaking filesystem paths | mitigate at Plan 02 boundary | HoldingsLoadError messages include `str(path)` for operator debugging at the server-side log. This plan's contract with Plan 02 is: loader exceptions are operator-facing; route responses are client-facing. |
| T-39-07 | Supply chain / code injection via CSV cells | `Holding` construction | accept | CSV cells are consumed only by `Decimal()` (numerically parsed) and stored as strings in `ticker`/`account`. No cell is ever interpolated into a log format, a Jinja template, or a Neo4j Cypher string. Stringly-typed ticker values (`"=SUM(A1)"` — CSV-injection payloads) are stored literally and pass through the advisory pipeline as opaque strings.
