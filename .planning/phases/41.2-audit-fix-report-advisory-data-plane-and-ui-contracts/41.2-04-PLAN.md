(1) Add `cycle_exists` helper to `src/alphaswarm/graph.py`. Place it near the other read methods (after `read_consensus_summary` is a natural location). Pattern (match the existing tx-helper style used in nearby methods):

```python
async def cycle_exists(self, cycle_id: str) -> bool:
    """Return True iff a (:Cycle {cycle_id}) node exists in Neo4j (D-06)."""
    try:
        async with self._driver.session(database=self._database) as session:
            return await session.execute_read(self._cycle_exists_tx, cycle_id)
    except Neo4jError as exc:
        raise Neo4jConnectionError(
            f"Failed to check cycle existence for {cycle_id}",
            original_error=exc,
        ) from exc

@staticmethod
async def _cycle_exists_tx(tx: AsyncManagedTransaction, cycle_id: str) -> bool:
    result = await tx.run(
        "MATCH (c:Cycle {cycle_id: $cycle_id}) RETURN count(c) > 0 AS exists",
        cycle_id=cycle_id,
    )
    record = await result.single()
    return bool(record["exists"]) if record else False
```
