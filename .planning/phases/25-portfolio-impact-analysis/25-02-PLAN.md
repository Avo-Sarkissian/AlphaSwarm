def build_react_system_prompt(*, include_portfolio: bool = False) -> str:
    """Build the ReACT system prompt with optional portfolio tool line.

    Per CONTEXT D-10 and RESEARCH Pitfall 4: the ReACT LLM only calls tools it
    knows exist. When --portfolio is provided at CLI, the portfolio_impact
    tool must appear in the available-tools list so the orchestrator will
    invoke it. Otherwise it must be absent (no phantom tool mentions).

    Args:
        include_portfolio: Append the portfolio_impact tool line when True.

    Returns:
        Complete ReACT system prompt string.
    """
    middle = _REACT_PROMPT_PORTFOLIO_LINE + "\n" if include_portfolio else ""
    return _REACT_PROMPT_HEADER + "\n" + middle + _REACT_PROMPT_FOOTER
