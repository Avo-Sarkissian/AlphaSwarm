```python
"""Portfolio CSV parsing and ticker-entity bridge for Phase 25.

Per CONTEXT.md D-05, holdings are loaded in-memory only and never persisted to
Neo4j or disk. Per D-06, D-07, D-08, the bridge is a case-insensitive substring
match between TICKER_ENTITY_MAP values and entity_name from read_entity_impact().
"""

from __future__ import annotations

import asyncio
import csv
import io
from pathlib import Path
from typing import TYPE_CHECKING

import structlog

if TYPE_CHECKING:
    from alphaswarm.graph import GraphStateManager

log = structlog.get_logger(component="portfolio")

# -----------------------------------------------------------------------------
# Constants
# -----------------------------------------------------------------------------

# Schwab CSV column names (verified from Individual-Positions-2026-04-09-154713.csv)
COLUMN_SYMBOL = "Symbol"
COLUMN_QTY = "Qty (Quantity)"
COLUMN_MKT_VAL = "Mkt Val (Market Value)"
COLUMN_ASSET_TYPE = "Asset Type"

_SENTINEL_SYMBOLS = frozenset({"Positions Total", "Cash & Cash Investments", ""})

# TICKER_ENTITY_MAP — per CONTEXT D-06/D-09. Maps ticker → list of canonical
# name substrings to match (case-insensitively) against entity_name values
# from GraphStateManager.read_entity_impact().
TICKER_ENTITY_MAP: dict[str, list[str]] = {
    "AAPL": ["Apple"],
    "AMZN": ["Amazon"],
    "ARM": ["Arm Holdings", "ARM Holdings", "Arm"],
    "ASML": ["ASML"],
    "AVGO": ["Broadcom"],
    "BYDDY": ["BYD"],
    "COHR": ["Coherent"],
    "DBX": ["Dropbox"],
    "HIMS": ["Hims & Hers", "Hims"],
    "HON": ["Honeywell"],
    "ISRG": ["Intuitive Surgical"],
    "LPL": ["LG Display"],
    "MRVL": ["Marvell"],
    "NIO": ["NIO", "Nio"],
    "NKE": ["Nike"],
    "NVDA": ["NVIDIA", "Nvidia"],
    "PLTR": ["Palantir"],
    "PYPL": ["PayPal", "Paypal"],
    "SCHW": ["Charles Schwab", "Schwab"],
    "SOFI": ["SoFi", "Sofi"],
    "TLN": ["Talen Energy", "Talen"],
    "TSLA": ["Tesla"],
    "TSM": ["Taiwan Semiconductor", "TSMC"],
    "VRT": ["Vertiv"],
    "VST": ["Vistra"],
}

# -----------------------------------------------------------------------------
# Helpers
# -----------------------------------------------------------------------------

def _parse_currency(raw: str) -> float:
    """Parse a Schwab-formatted currency string to float.

    Handles: "$26,416.56 " (trailing space), "($4,472.00)" (parens-negative),
    "1,000.00" (no $), "".
    """
    s = raw.strip()
    if not s:
        return 0.0
    # Parens => negative
    if s.startswith("(") and s.endswith(")"):
        s = "-" + s[1:-1]
    s = s.replace("$", "").replace(",", "").replace(" ", "")
    return float(s)


def _parse_quantity(raw: str) -> float:
    """Parse a Schwab-formatted quantity string to float (handles thousands-sep)."""
    s = raw.strip().replace(",", "")
    if not s:
        return 0.0
    return float(s)

# -----------------------------------------------------------------------------
# Public API
# -----------------------------------------------------------------------------

def parse_schwab_csv(path: Path) -> dict[str, dict]:
    """Parse a Schwab Individual-Positions CSV export into an in-memory holdings dict.

    Per CONTEXT.md D-01..D-05:
      - Skips the 2 metadata rows at the top (row 3 = header)
      - Filters to Asset Type == "Equity"
      - Strips currency formatting ($, commas, spaces, parens)
      - Never persists anything

    Args:
        path: Path to the Schwab Individual-Positions CSV file.

    Returns:
        dict keyed by ticker symbol. Each value is a dict with keys
        {ticker, shares, market_value}. Empty dict if no Equity rows.
    """
    text = path.read_text(encoding="utf-8")
    lines = text.splitlines()
    if len(lines) < 3:
        log.warning("schwab_csv_too_short", path=str(path), line_count=len(lines))
        return {}
    body = "\n".join(lines[2:])
    reader = csv.DictReader(io.StringIO(body))
    holdings: dict[str, dict] = {}
    for row in reader:
        symbol = (row.get(COLUMN_SYMBOL) or "").strip()
        if symbol in _SENTINEL_SYMBOLS:
            continue
        asset_type = (row.get(COLUMN_ASSET_TYPE) or "").strip()
        if asset_type != "Equity":
            continue
        try:
            shares = _parse_quantity(row.get(COLUMN_QTY) or "")
            market_value = _parse_currency(row.get(COLUMN_MKT_VAL) or "")
        except ValueError as exc:
            log.warning(
                "schwab_row_parse_failed",
                ticker=symbol,
                raw_qty=row.get(COLUMN_QTY),
                raw_mkt_val=row.get(COLUMN_MKT_VAL),
                error=str(exc),
            )
            continue
        holdings[symbol] = {
            "ticker": symbol,
            "shares": shares,
            "market_value": market_value,
        }
    return holdings


async def parse_schwab_csv_async(path: Path) -> dict[str, dict]:
    """Async wrapper over parse_schwab_csv — runs file read + parse in a thread."""
    return await asyncio.to_thread(parse_schwab_csv, path)


async def build_portfolio_impact(
    holdings: dict[str, dict],
    gm: "GraphStateManager",
    cycle_id: str,
) -> dict:
    """Bridge parsed holdings against swarm entity_impact results.

    Per CONTEXT.md D-07, D-08, D-11:
      - Fetches entity_impact for the cycle
      - Performs case-insensitive substring match using TICKER_ENTITY_MAP
      - Returns {matched_tickers, gap_tickers, coverage_summary}

    Args:
        holdings: In-memory dict from parse_schwab_csv().
        gm: GraphStateManager for read_entity_impact().
        cycle_id: Simulation cycle id to query.

    Returns:
        dict with keys matched_tickers (list), gap_tickers (list),
        coverage_summary (dict).
    """
    entity_results = await gm.read_entity_impact(cycle_id)

    matched: list[dict] = []
    gaps: list[dict] = []

    for ticker, pos in holdings.items():
        entity_substrings = TICKER_ENTITY_MAP.get(ticker, [])
        found_entity: dict | None = None
        for substring in entity_substrings:
            sub_lower = substring.lower()
            for entity in entity_results:
                entity_name = entity.get("entity_name", "")
                if sub_lower in entity_name.lower():
                    found_entity = entity
                    break
            if found_entity is not None:
                break

        market_value = float(pos["market_value"])
        market_value_display = f"${market_value:,.2f}"

        if found_entity is not None:
            buy = int(found_entity.get("buy_mentions", 0))
            sell = int(found_entity.get("sell_mentions", 0))
            hold = int(found_entity.get("hold_mentions", 0))
            counts = {"BUY": buy, "SELL": sell, "HOLD": hold}
            majority_signal = max(counts, key=lambda k: counts[k])
            total = sum(counts.values())
            confidence = (counts[majority_signal] / total) if total else 0.0
            matched.append({
                "ticker": ticker,
                "shares": float(pos["shares"]),
                "market_value": market_value,
                "market_value_display": market_value_display,
                "signal": majority_signal,
                "confidence": round(confidence, 3),
                "entity_name": found_entity.get("entity_name", ""),
                "avg_sentiment": round(float(found_entity.get("avg_sentiment", 0.0)), 3),
                "mention_count": int(found_entity.get("mention_count", 0)),
            })
        else:
            gaps.append({
                "ticker": ticker,
                "shares": float(pos["shares"]),
                "market_value": market_value,
                "market_value_display": market_value_display,
            })

    total_holdings = len(holdings)
    covered = len(matched)
    coverage_pct = round(covered / total_holdings * 100, 1) if total_holdings else 0.0

    return {
        "matched_tickers": matched,
        "gap_tickers": gaps,
        "coverage_summary": {
            "covered": covered,
            "total_equity_holdings": total_holdings,
            "coverage_pct": coverage_pct,
        },
    }
```
